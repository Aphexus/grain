test opBinary(string op)(Variable ...)

1 import mir.ndslice;
2 import numir;
3 import std.stdio;
4 static foreach (op; ["+", "*", "-", "/"]) {
5     {
6         auto a = uniform!float(3, 2).slice.variable(true);
7         auto b = uniform!float(3, 2).slice.variable(true);
8         // this is equivalent to `a + b` if op == "+"
9         auto c = a.opBinary!op(b);
10         // this is equivalent to `a.sliced.slice + b.sliced.slice` if op == "+"
11         auto e = a.sliced.slice.opBinary!op(b.sliced.slice);
12         assert(approxEqual(c.sliced, e));
13 
14         auto gc = uniform!float(3, 2).slice.variable(true);
15         auto ugc = UntypedVariable(gc);
16         c.backward(&ugc);
17 
18         version (grain_cuda) {
19             auto da = a.to!DeviceStorage;
20             auto db = b.to!DeviceStorage;
21             auto dc = da.opBinary!op(db);
22             assert(approxEqual(dc.to!HostStorage.sliced, c.sliced));
23 
24             import grain.cuda : zero_;
25             da.grad.zero_();
26             db.grad.zero_();
27             auto dugc = UntypedVariable(gc.to!DeviceStorage);
28             dc.backward(&dugc);
29             assert(approxEqual(da.to!HostStorage.gradSliced, a.gradSliced));
30         }
31     }
32 }
